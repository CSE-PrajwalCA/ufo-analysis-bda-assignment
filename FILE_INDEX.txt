🔌 REST API ENDPOINTS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

GET /
  Serves the dashboard HTML page
  Response: HTML (index.html)

GET /distinct/country
  Returns unique country values
  Response: JSON {"success": true, "values": [...]}

GET /distinct/state
  Returns unique state values
  Response: JSON {"success": true, "values": [...]}

GET /distinct/shape
  Returns unique shape values
  Response: JSON {"success": true, "values": [...]}

POST /filter
  Returns filtered UFO sighting records
  Request Body: {
    "country": "us",
    "state": "tx",
    "shape": "circle",
    "yearRange": [1949, 2010],
    "durationRange": [0, 10000]
  }
  Response: JSON {"success": true, "count": N, "data": [...]}

POST /summary
  Returns summary statistics
  Response: JSON {"success": true, "stats": {...}}

POST /export/csv
  Downloads filtered data as CSV
  Response: CSV file download
